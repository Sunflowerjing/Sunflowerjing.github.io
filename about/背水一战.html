    L = L.__proto__;    //L为stu._proto_，也随着prototype的改变而指向了per
 
    while (true) {    //执行循环
          if (L === null)  //不通过
              return false;   
          if (O === L)    //判断： Student.prototype ===stu._proto_？
              return true;  //此时，两方都指Person的实例对象per，所以true
          L = L.__proto__;                   
    }

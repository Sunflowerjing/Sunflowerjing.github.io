# 大数相加



```js
function add(a,b){
    var la = a.length - 1, lb = b.length -1;
    var sum = 0, result = '';
    while(la >= 0 || lb >= 0 || sum > 0){
        if(la > 0) sum += parseInt(a[la--]);
        if(lb > 0) sum += parseInt(b[lb--]);
        result = (sum%10) + result;
        sum = Math.floor(sum/10);
    }
    return result;
}
```
